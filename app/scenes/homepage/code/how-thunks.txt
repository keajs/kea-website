const delay = ms => new Promise(resolve => window.setTimeout(resolve, ms))

const incrementerLogic = kea({
  actions: () => ({
    increase: true
  }),
  
  reducers: ({ actions }) => ({
    counter: [0, {
      [actions.increase]: (state, payload) => state + 1
    }]
  }),

  thunks: ({ actions, dispatch, getState }) => ({
    asyncIncrease: async (ms) => {
      await delay(ms)
      actions.increase()
    }
  })
})
